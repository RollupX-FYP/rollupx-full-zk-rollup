\headerbox{7. Benchmark Design}{name=refs,column=0,below=background,span=1}{
\begin{itemize}[leftmargin=0.75em,itemsep=0.08em,topsep=0pt,parsep=0pt]
    \item \textbf{Workloads:} Synthetic steady-state, skewed, and bursty transfer patterns.
    \item \textbf{Controls:} Batch policy, sequencing policy, DA mode, and proof-delay setting.
    \item \textbf{Metrics:} Committed TPS, latency, cost/tx, DA bytes, and finalization time.
    \item \textbf{Scope:} Relative configuration comparison, not production mainnet performance.
\end{itemize}
